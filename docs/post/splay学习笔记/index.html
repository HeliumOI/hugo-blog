void insert(int x) {
    int cur = root, p = 0;
    while(cur != 0 && node[cur].v != x) {
        p = cur;
        cur = node[cur].ch[x < node[cur].v ? 0 : 1];

        node[cur].cnt++;

        cur = ++cnt;

    splay(cur);
}
