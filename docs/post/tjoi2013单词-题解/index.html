#include<bits/stdc++.h>
using namespace std;

const int MAXN = 1000000 + 5;
const int MAXM = 200 + 5;

string s[MAXM];
bool vis[MAXN];
int t[MAXN][26], sum[MAXN], cnt[MAXN], ed[MAXN], ans[MAXN];
int fail[MAXN];
int tot;

int insert(string& s) {
    int p = 0;
    for(int i = 0; i < s.length(); i++) {
        if(t[p][s[i] - 'a'] == 0) {
            t[p][s[i] - 'a'] = ++tot;
        }
        p = t[p][s[i] - 'a'];
        cnt[p]++;
    }
    return p;
}

void getFail() {
    queue<int> q;
    for(int i = 0; i < 26; i++) {
        if(t[0][i]) {
            q.push(t[0][i]);
        }
    }
    while(!q.empty()) {
        int u = q.front(); q.pop();

        for(int i = 0; i < 26; i++) {
            if(t[u][i]) {
                fail[t[u][i]] = t[fail[u]][i];
                q.push(t[u][i]);
            } else {
                t[u][i] = t[fail[u]][i];
            }
        }
    }
}

void query(string& s) {
    int p = 0;

    for(int i = 0; i < s.length(); i++) {
        p = t[p][s[i] - 'a'];

        for(int j = p; j != 0 && !vis[p]; j = fail[j]) {
            ans[j] += cnt[p];
        }
        vis[p] = 1;
    }
}

int main() {
    int n;

    cin >> n;
    for(int i = 1; i <= n; i++) {
        cin >> s[i];
        ed[i] = insert(s[i]);
    }
    getFail();
    for(int i = 1; i <= n; i++) {
        query(s[i]);
    }
    for(int i = 1; i <= n; i++) {
        cout << ans[ed[i]] << endl;
    }

    return 0;
}
